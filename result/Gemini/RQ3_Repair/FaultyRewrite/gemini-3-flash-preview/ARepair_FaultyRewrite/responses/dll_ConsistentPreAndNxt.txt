}

pred ConsistentPreAndNxt() {
  all n1, n2: Node | n1 != n2 => (n1.nxt = n2 => n2.pre = n1) and (n2.pre = n1 => n1.nxt = n2)
}

pred ConsistentPreAndNxt() {

}
